600 {

603 #endif
604  // increase/decrease size of image area
605  double newImageWidth = zoomAmount + m_imageRect.width();
606  double newImageHeight = zoomAmount + m_imageRect.height();
607 
608  // catch image bigger than a pixel
609  if(newImageWidth<0 || newImageHeight<0)
610  return;
611 
612  // get ratio of window to new image size
613  double ratioW = (newImageWidth / (double)m_displayRect.width());
614  double ratioH = (newImageHeight / (double)m_displayRect.height());
615 
616  // get clicked coordinate relative to window size
617  double imageScaleW = ((double)windowClick.x())/((double)m_displayRect.width());
618  double imageScaleH = ((double)windowClick.y())/((double)m_displayRect.height());
619 
620  // get pixel clicked in image space
621  QPoint imageClickPixel = QPoint(((int)m_imageRect.left() + m_imageRect.width()*imageScaleW),
622  ((int)m_imageRect.top() + m_imageRect.height()*imageScaleH));
623 
624  // get the window clicked position
625  double clickWindowPositionX = (double)windowClick.x();
626  double clickWindowPositionY = (double)windowClick.y();
627 
628  int left = imageClickPixel.x() - (int)(clickWindowPositionX * ratioW);
629  int right = imageClickPixel.y() - (int)(clickWindowPositionY * ratioH);
630 
631  // adjust image boundaries to keep clicked point in same position
632 
633  // new size
634  QRect r = m_imageRect;
635 
636  r.setLeft(left);
637  r.setTop(right);
638 
639  // resize rect to size of zoomed area
640  r.setWidth(newImageWidth);
641  r.setHeight(newImageHeight);
642 
643  // catch image smaller than window
